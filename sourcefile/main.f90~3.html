#include "precompilerdefinitions"
program crystal_structure_info
!!{!src/crystal_structure_info/manual.md!}
!
! Prints general info about the lattice, high symmetry points and so on.
!
use constants
use helpers
use dump_data
use options
use geometryfunctions
use type_crystalstructure
use type_qpointmesh
use type_symmetryoperation
!
implicit none
! for normal phonon things
type(lo_opts) :: opts
type(lo_crystalstructure) :: uc
type(lo_bandstructure) :: bs
!
real(flyt), dimension(3) :: v0,v1
real(flyt) :: tstart,f0 
integer :: i,j,u
character(len=1000) :: opf
!
! Get the command line arguments
!
tstart=walltime()
call opts%parse()
!
! Read unitcell POSCAR
!
call uc%readfromfile('infile.ucposcar',verbosity=opts%verbosity)
!
write(*,*) 'Info about the crystal structure:'
write(*,*) '... I believe the basis is ',trim(uc%bravaislatticelongname),' (',trim(uc%bravaislattice),') with the basis vectors'
write(*,"(1X,'a1:',3(2X,F15.11))") uc%fullbasis(1,:)
write(*,"(1X,'a2:',3(2X,F15.11))") uc%fullbasis(2,:)
write(*,"(1X,'a3:',3(2X,F15.11))") uc%fullbasis(3,:)
write(*,*) '... and the atoms at these positions (in fractional coordinates)'
do i=1,uc%na
    write(*,'(1X,A6,3(2x,F15.11))') uc%atomic_symbol(uc%species(i)),uc%r(:,i)
enddo
!
! Print what the conventional basis is
!
f0=lo_frobnorm(uc%fullbasis-uc%unique_conventional_basis)
if ( f0 .gt. lo_tol ) then
    write(*,*) 'I believe this is the conventional lattice:'
    write(*,"(1X,'a1:',3(2X,F15.11))") uc%unique_conventional_basis(1,:)
    write(*,"(1X,'a2:',3(2X,F15.11))") uc%unique_conventional_basis(2,:)
    write(*,"(1X,'a3:',3(2X,F15.11))") uc%unique_conventional_basis(3,:)
endif
!
! It might not agree with my canonical form.
!
!f0=lo_frobnorm(uc%fullbasis-uc%unique_primitive_basis)
!if ( f0 .gt. lo_tol ) then
!    write(*,*) ' '
!    write(*,*) 'I like this primitive basis better:'
!    write(*,"(1X,'a1:',3(2X,F16.12))") uc%unique_primitive_basis(1,:)
!    write(*,"(1X,'a2:',3(2X,F16.12))") uc%unique_primitive_basis(2,:)
!    write(*,"(1X,'a3:',3(2X,F16.12))") uc%unique_primitive_basis(3,:)
!    write(*,*) '... with the atoms at these positions (in fractional coordinates)'
!    m0=uc%unique_conventional_basis
!    call lo_invertmatrix_lapack(m0)
!    !
!    ! Convert the coordinates
!    !
!    allocate(dum(3,uc%na))
!    do i=1,uc%na
!        ! get the current coordinate
!        v0=uc%r(:,i)
!        v0=matmul(uc%permutation_to_unique,v0)
!        do j=1,3
!            if ( v0(j) .lt. 0.0_flyt ) v0(j)=v0(j)+1.0_flyt
!            if ( v0(j) .gt. 1.0_flyt ) v0(j)=v0(j)-1.0_flyt
!            if ( abs(v0(j)) .lt. 1E-10_flyt ) v0(j)=0.0_flyt
!        enddo
!        dum(:,i)=v0
!        write(*,'(1X,A6,3(2x,F15.11))') uc%atomic_symbol(uc%species(i)),dum(:,i)
!    enddo
!    !
!endif
!
! 
!
write(*,*) ' '
write(*,*) 'Available high symmetry points:'
write(*,*) '(the ones called NP are in fact high symmetry points, but noone has bothered to name them)'
write(*,*) '      Cartesian coordinates                        Fractional coordinates'
do i=1,uc%irrw%nnodes
    v0=uc%irrw%r(:,i)
    v1=matmul(v0,uc%invrecbasis)
    do j=1,3
        if ( abs(v0(j)) .lt. lo_tol ) v0(j)=0.0_flyt
        if ( abs(v1(j)) .lt. lo_tol ) v1(j)=0.0_flyt
    enddo
    write(*,"(1X,I2,1X,A2,3(1X,F13.10),3X,3(1X,F13.10))") i,uc%irrw%label(i),v0,v1
enddo
!
call uc%write_bz_to_hdf5('outfile.brillouin_zone.hdf5')

!call uc%irrw%dump_to_matlab('ml_wedge.m','wedge_')
!call uc%bz%dump_to_matlab('ml_bz.m','bz_')
!
call bs%standardpath(uc)
bs%npts=100
write(*,*) ' '
write(*,*) 'Per default, the following path will be used:'
do i=1,bs%npath
    write(*,*) trim(bs%symb_q_start(i)),' -> ',trim(bs%symb_q_end(i))
enddo
write(*,*) 'It is written in "outfile.qpoints_dispersion", modify and'
write(*,*) 'copy it to "infile.qpoints_dispersion" if you want'
!
! Dump a path file
!
u=open_file('out','outfile.qpoints_dispersion')
    opf="(A5,22X,' ! Bravais lattice type')"
    write(u,opf) uc%bravaislattice
    opf="(I5,22X,' ! Number of points on each path')"
    write(u,opf) 100
    opf="(I5,22X,' ! Number paths between special points')"
    write(u,opf) bs%npath
    opf="(A3,1X,A3,20X,' ! Starting and ending special point')"
    write(u,opf) bs%symb_q_start(1),bs%symb_q_end(1)
    opf="(A3,1X,A3,20X,' !')"
    do i=2,bs%npath
        write(u,opf) bs%symb_q_start(i),bs%symb_q_end(i)
    enddo
close(u)

write(*,*) ' '
write(*,*) 'The Brillouin zone was written in matlab format to'
write(*,*) 'ml_bz.m and ml_wedge.m for plotting.'
write(*,*) ' '
!
if ( opts%printsymmetry ) then
    write(*,*) ' Symmetry operations:'
    do i=1,uc%sym%n
        write(*,*) 'Operation ',trim(int2char(i)),' ',uc%sym%op(i)%whatkind
            write(*,*) '    Axis:',real(uc%sym%op(i)%axis)
            if ( uc%sym%op(i)%fold .gt. 0 ) write(*,*) '   Angle:',360.0_flyt/uc%sym%op(i)%fold
            !write(*,*) '    FMAP:',uc%sym%op(i)%fmap
        write(*,*) ' ... in Cartesian coordinates'
        do j=1,3
            write(*,"(6(3X,F15.11))") uc%sym%op(i)%m(j,:)
        enddo
        write(*,*) ' ... in fractional coordinates'
        do j=1,3
            write(*,"(6(3X,F15.11))") uc%sym%op(i)%fm(j,:)
        enddo
    enddo
    !
endif
!
write(*,*) 'all done'
write(*,*) ' '

end program
